\begin{estrofa}

\numero{1} \textparagraph\  Coratge meu apendres fort molt tart

\numero{2} No piados de tots los quit sostenen

\numero{3} Larmay lo cors adepartir se venen

\numero{4} Pertu ser flach lo cors de viures fart

\numero{5} Mos hulls no son libers fer son offir

\numero{6} Mon pas es tolt mal lengua nomprofita

\numero{7} Edeaco lavergonyas delita

\numero{8} Com so plagat de tan bestial vici

\end{estrofa}



\begin{estrofa}

\numero{9} [85] Paor me sent gran suor me comence

\numero{10} Surtint mon cor los pits me cuyden rompre

\numero{11} Nom trob esforc per vergonya corompre

\numero{12} Esser no pot masperanca por venca

\numero{13} No puch mostrar lo secret demapensa

\numero{14} E vanament epor dela resposta

\numero{15} \interlin{Lomeu dubtar major duptem acosta}

\numero{16} Ffemenil gest ardiment me defensa

\end{estrofa}



\begin{estrofa}

\numero{17} Alguns an dit que vergonya nos troba

\numero{18} Mas yompuch fer daquella testimonj

\numero{19} De vista no semblant es al dimonj

\numero{20} Part demos senys eparlar me deroba

\numero{21} Dona sentir de si al\sout{s} guns forts actes

\numero{22} Segons de molts auem hoydes gestes

\numero{23} Crehen lo tals quj descolen ses festes

\numero{24} Senyor es meu per amigables pactes

\end{estrofa}



\begin{estrofa}

\numero{25} E sil pogues lobedienca tolre

\numero{26} Que per son colp la mia mor jau morta

\numero{27} Car no parent les altramor estorta

\numero{28} No se quin deu damnat sant me fa colre

\numero{29} Yo agrasforc demetre tost enobre

\numero{30} Lo que no gos mostrar vna parenca

\numero{31} Ab gran esforc vergonyam fa temenca

\numero{32} Si ardiment vnpoch enmj no sobre

\end{estrofa}



\begin{estrofa}

\numero{33} [86] Entots aquells hon gran amor no penja

\numero{34} Son gin no pot de vergonya nesforca

\numero{35} Amj epochs ason voler nos forca

\numero{36} Nostre voler nostres enemichs venya

\numero{37} Aquest voler desig hi amor sostenen

\numero{38} Causa daquests sou vos aquj sesguarden

\numero{39} Creure no puch vostres sentiments tarden

\numero{40} Endescobrir los quals que pervos venen

\end{estrofa}
