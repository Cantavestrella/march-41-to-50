\begin{estrofa}

\numero{1} C\textsc{o}ratje meu / a pendre'sfor\c{c} molt tart

\numero{2} no piados / de tots los quit sostenen

\numero{3} l'arma y lo cos / a departirse venen

\numero{4} per tu ser flach / lo cos de viure es fart.

\numero{5} Mos vlls no son / liberts ason offi\c{c}i

\numero{6} mon pas es tolt / ma lengua nom profita

\numero{7} e de a\c{c}o / la vergonya s'delita

\numero{8} com so plagat / de tant vergonyos vici.

\end{estrofa}



\begin{estrofa}

\numero{9} \textparagraph\  Pahor me sent / gran suor me comen\c{c}a

\numero{10} surtint mon cor / los pits me cuy da rompre

\numero{11} nom trob esfor\c{c} / per vergonya corrompre

\numero{12} esser no pot / speran\c{c}a pahor ven\c{c}a.

\numero{13} No puch mostrar / lo secret de ma pensa

\numero{14} e vanament / he por de la resposta

\numero{15} lo meu duptar / major dupte m'acosta

\numero{16} feminil gest / ardiment me defensa.

\end{estrofa}



\begin{estrofa}

\numero{17} \textparagraph\  Alguns han dit / que vergonya nos troba

\numero{18} mas yo puch fer / de aquella testimoni

\numero{19} de vista no / semblant es al demoni

\numero{20} part de mos senys / e parlar me derroba.

\numero{21} Dona'sentit / desi alguns forts acctes

\numero{22} segons de molts / hauem hoydes gestes

\numero{23} crehent los tals / qui descolen les festes

\numero{24} senyor es meu / Amor fermals contractes.

\end{estrofa}



\begin{estrofa}

\numero{25} [83v] \textparagraph\  E sil'posques / / la obedien\c{c}a tolrre

\numero{26} que per son colp / la mia'mor jau morta

\numero{27} car no parent / l'es altra'mor estorta

\numero{28} no se quin Deu / damnat sant me fa colrre.

\numero{29} Yo aguera's for\c{c} / de metre tost en obra

\numero{30} lo que no gos / mostrar vna'paren\c{c}a

\numero{31} ab gran esfor\c{c} / vergonyam'fa temen\c{c}a

\numero{32} si qu'ardiment / vn poch en mi no sobra.

\end{estrofa}



\begin{estrofa}

\numero{33} \textparagraph\  Entots aquells / hon gran Amor no penja

\numero{34} son giny no pot / de vergonya ne for\c{c}a

\numero{35} ami e pochs / a son voler nos for\c{c}a

\numero{36} lo meu voler / los meus enemichs venja.

\numero{37} Aquest voler / desig y Amor sostenen

\numero{38} causa d'aquests / sou vos aqui se'sguarden

\numero{39} creure no puch / vostres sentiments tarden

\numero{40} en descobrir / los mals que per vos venen.

\end{estrofa}
