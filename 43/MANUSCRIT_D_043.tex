\begin{estrofa}

\numero{1} Coratge meu / a pendre \sout{e}sfor\c{c} molt tart

\numero{2} no piados / detots los quit sostenen

\numero{3} larma y lo cos / adepartir sevenen

\numero{4} pertu ser flach / lo cos deviures fart

\numero{5} Mos vlls no son / liberts a \sout{fer} son ofici

\numero{6} monpas es tolt / ma lengua nomprofita

\numero{7} edea\c{c}o / lavergonyas delita

\numero{8} com so plagat / detan vergonyos vici

\end{estrofa}



\begin{estrofa}

\numero{9} Pahor me sent / gran suor mecomen\c{c}a

\numero{10} surtint moncor / los pits me cuydarompre

\numero{11} nomtrob esfor\c{c} / pervergonya corrompre

\numero{12} esser no pot \interlin{ma} esperan\c{c}a por uen\c{c}a

\numero{13} No puch mostrar / lo secret de mapensa

\numero{14} evanament / hepordelaresposta

\numero{15} lo meu duptar / major duptem acosta

\numero{16} femenil gest / ardiment me defensa

\end{estrofa}



\begin{estrofa}

\numero{17} Alguns handit / quevergonya nos troba

\numero{18} mas yo puch fer / d'aquellatestimoni

\numero{19} devista no / semblant es al dimoni

\numero{20} part de mos senys / eparlar me derroba

\numero{21} Dona sentir / de si alguns forts actes

\numero{22} segons demolts / hauem hoydes gestes

\numero{23} crehent los tals / qui descolen les festes

\numero{24} sen\textit{y}or es meu / amor fermal's contractes

\end{estrofa}



\begin{estrofa}

\numero{25} Esil posques / lobedien\c{c}\sout{i}a tolre

\numero{26} que per soncolp / lamia'mor jau morta

\numero{27} car no parent / l'es altra'mor estorta

\numero{28} no se quin deu / damnat sanct me fa colre

\numero{29} [39v] Yo agra'sfor\c{c} / de metre tost en obra

\numero{30} lo que nogos / mostrar vnaparen\c{c}a

\numero{31} abgran esfor\c{c} / vergonyam fa temen\c{c}a

\numero{32} siqu'ardiment\footnote{siqu'a:\textit{ escrit sobre un paraula raspada i
il$\cdot{}$legible. Al marge esquerre, hi ha una annotaci\'{o} tamb\'{e}
il$\cdot{}$legible.}} /vn poch en mi no sobra

\end{estrofa}



\begin{estrofa}

\numero{33} Entots aquells / on gran amor no penja

\numero{34} son giny no pot / devergonya ne for\c{c}a

\numero{35} a mi e pochs / a sonvoler nos for\c{c}a

\numero{36} lo meu voler / los meus enemichs venja

\numero{37} Aquest voler / desig y amor sostenen

\numero{38} causa d'aquests / souvos aqui sesguarden

\numero{39} creure no puch / vostres sentiments tarden

\numero{40} endescobrir / los mals quepervos venen

\end{estrofa}
