\poema{(e 68) [43]}\\

\begin{rubrica}

DE AMOR CANT. LXVIII.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} C\textsc{o}ratje meu / a pendre'sfor\c{c} molt tart

\numero{2} no piados / de tots los quit sostenen

\numero{3} l'arma y lo cos / a departir se venen

\numero{4} per tu ser flach / lo cos de viure es fart.

\numero{5} [77v] Mos vlls no son / liberts a son offici

\numero{6} mon pas es tolt / ma lengua nom profita

\numero{7} e de a\c{c}o / la vergonya's delita

\numero{8} com so plagat / de tan vergonyos vici.

\end{estrofa}



\begin{estrofa}

\numero{9} Pahor me sent / gran suor me comen\c{c}a

\numero{10} surtint mon cor / los pits me cuyda rompre

\numero{11} nom trob esfor\c{c} / per vergonya corrompre

\numero{12} esser no pot / speran\c{c}a por ven\c{c}a.

\numero{13} No puch mostrar / los ecret de ma pensa

\numero{14} e vanament / he por dela resposta

\numero{15} lo meu duptar / major dubte m'acosta

\numero{16} feminil gest / ardiment me defensa.

\end{estrofa}



\begin{estrofa}

\numero{17} Alguns han dit / que vergonya nos troba

\numero{18} mas yo puch fer / d'aquella testimoni

\numero{19} de vista no / semblant es al dimoni

\numero{20} part de mos senys / e parlar me derroba.

\numero{21} Don'asentir / de si alguns forts actes

\numero{22} segons de molts / hauem hoy des gestes

\numero{23} crehent los tals / qui descolen les festes

\numero{24} senyor es meu / Amor ferm'als contractes.

\end{estrofa}



\begin{estrofa}

\numero{25} E sil'posques / l'obedien\c{c}a tolrre

\numero{26} que per son colp / la mi'amor jau morta

\numero{27} car no parent / l'es altr'amor estorta

\numero{28} [78r] no se quin Deu / damnat sant me fa colrre.

\numero{29} Yo'guera'sfor\c{c} / de metre tost en obra

\numero{30} lo que no gos / mostrar vna'paren\c{c}a

\numero{31} ab gran esfor\c{c} / vergonyam'fa temen\c{c}a

\numero{32} si qu'ardiment / vn poch en mi no sobra.

\end{estrofa}



\begin{estrofa}

\numero{33} En tots aquells / hon gran Amor no penja

\numero{34} son giny no pot / de vergonya ne for\c{c}a

\numero{35} ami e pochs / a son voler nos for\c{c}a

\numero{36} lo meu voler / los meus enemichs venja.

\numero{37} Aquest voler / desig y Amor sostenen

\numero{38} causa d'aquests / sou vos aqui se'sguarden

\numero{39} creure no puch / vostres sent\'{\i}ments tarden

\numero{40} en descobrir / los mals que per vos venen.

\end{estrofa}

\numero{41} O foll Amor / aquells dolor sostenen

\numero{42} qui cerquen fi / hon la fi no pot esser

\numero{43} de llur treball / no s'mostra res en esser

\numero{44} e son aquests / los qui de vos s'encenen.
