\poema{(E 24) [43]}\\

\begin{rubrica}

Moss\textit{en} ausias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Coratge meu / ha pendre sfor\c{c} molt tart

\numero{2} no piados / de tots los quit sostenen

\numero{3} larmay lo cos / adepartir se venen

\numero{4} per tu ser flach / lo cos de viures fart

\numero{5} Mos vlls no son / lliberts fer son offi\c{c}i

\numero{6} mon pas es tolt / ma llengua nom profita

\numero{7} e de ay\c{c}o / la vergonyas delita

\numero{8} com so plagat / de tan vergonyos vi\c{c}i

\end{estrofa}



\begin{estrofa}

\numero{9} [34v] Pahor me sent / gran suor me comen\c{c}a

\numero{10} surtint mon cor / los pits me cuyda rompre

\numero{11} nom trob sfor\c{c} / per vergonya corrompre

\numero{12} esser no pot / masperan\c{c}a por ven\c{c}a

\numero{13} No pusch mostrar / lo secret de ma pensa

\numero{14} e vanament / he por dela resposta

\numero{15} lo meu dubtar / maior dubte macosta

\numero{16} femenil gest / ardiment me deffensa

\end{estrofa}



\begin{estrofa}

\numero{17} Alguns han dit / que vergonya nos troba

\numero{18} mas yom pusch fer / daquella testimoni

\numero{19} de vista no / ques semblant al dimoni

\numero{20} part de mos senys / e parlar me derroba

\numero{21} Dona sentir / de si alguns forts actes

\numero{22} segons de molts / hauem hoydes gestes

\numero{23} crehen la tals / qui descolen ses festes

\numero{24} senyor es meu / amor fermals contractes

\end{estrofa}



\begin{estrofa}

\numero{25} E sil posques / lobedien\c{c}a tolre

\numero{26} que per son colp / la mia mor jau morta

\numero{27} car no parent / les altra mor estorta

\numero{28} nose quin deu / damnat sant me fa colre

\numero{29} Yo agrasfor\c{c} / de metre tost en obra

\numero{30} lo que no gos / mostrar en apparen\c{c}a

\numero{31} ab gran sfor\c{c} / vergonyam fa temen\c{c}a

\numero{32} si quardiment / vn poch en mi no sobra

\end{estrofa}



\begin{estrofa}

\numero{33} [35r] En tots aquells / hon gran amor no penja

\numero{34} son giny no pot / de vergonya ne for\c{c}a

\numero{35} ami e pochs / ason voler nos for\c{c}a

\numero{36} nostre voler / nostres enemichs venja

\numero{37} Aquest voler / desig yamor sostenen

\numero{38} causa daquests / sou vos aqui sesguarden

\numero{39} creure no pusch / vostres sentiments tarden

\numero{40} en descobrir / los mals que per vos venen

\end{estrofa}
