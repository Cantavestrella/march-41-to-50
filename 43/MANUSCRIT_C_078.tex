\begin{estrofa}

\numero{1} [104v] Coratje meu / apendre sfor\c{c} molt tart

\numero{2} no piados / detots los quit'sostenen

\numero{3} l'arma y lo cos / ade partir se uenen

\numero{4} per tu ser flach / lo cors de uiure es fart

\numero{5} mos ulls / no son / liberts a son offici

\numero{6} mon pas e tolt / malengua nom profita

\numero{7} e de a\c{c}o / la uergonya s'delita

\numero{8} com so plagat / de tant uergonyos uici.

\end{estrofa}



\begin{estrofa}

\numero{9} Pahor me sent / gran suor me comen\c{c}a

\numero{10} sortint mon cor / los pits me cuida rompre

\numero{11} nom trobe sfor\c{c} / per uergonya corrompre

\numero{12} esser no pot / esperan\c{c}a p'or uen\c{c}a

\numero{13} no puch mostrar / lo secret de ma pensa

\numero{14} e uanament / he p'or de la resposta

\numero{15} lo meu dubtar\footnote{\textit{Es pot llegir} dubtar o duptar.} / major dubte
m'acosta

\numero{16} femenil gest / ardiment me defensa

\end{estrofa}



\begin{estrofa}

\numero{17} Alguns han dit / que uergonya nos troba

\numero{18} mas yo puch fer / d'aquella testimoni

\numero{19} de uista no / semblant es al demoni

\numero{20} part de mos senys / e parlar me deroba

\numero{21} [105r] dona sentir desi / alguns forts actes

\numero{22} segons de molts / hauem ohydes gestes

\numero{23} crehent los tals / qui descolen les festes

\numero{24} senyor es meu / amor fermal's contractes.

\end{estrofa}



\begin{estrofa}

\numero{25} Esil posques / l'obedien\c{c}a tolre

\numero{26} que per son colp / la mi amor jau morta

\numero{27} car no parent / l'es altr'amor estorta

\numero{28} nose quin Deu / damnat sant me fa colre

\numero{29} yo ag'ra'sfor\c{c} / de metre test enobra

\numero{30} lo que no gos / mostrar una paren\c{c}a

\numero{31} ab gran esfor\c{c} / uergonya m'fa temen\c{c}a

\numero{32} si qu'rdiment / un poch en mi no sobra.

\end{estrofa}



\begin{estrofa}

\numero{33} E tots aquels / hon gran amor no penja

\numero{34} son giny no pot / de uergonya ni for\c{c}a

\numero{35} ami e pochs / a son uoler nos for\c{c}a

\numero{36} lo meu uoler / los meus enemichs uenja

\numero{37} aquest uoler / desig, y amor soste

\numero{38} causa d'aquests / sou uos aqu\'{\i} sesguarden

\numero{39} creure no puch / uostres sentiments tarden

\numero{40} en descobrir / los mals que per uos uenen.

\end{estrofa}
