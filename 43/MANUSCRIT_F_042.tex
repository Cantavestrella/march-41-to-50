\begin{estrofa}

\numero{1} [41r] Paor me sent gran suor me comenca

\numero{2} Surtint mon cor lo pits me cuyda rompre

\numero{3} Nom trob esforc per uergonya corrompre

\numero{4} Esser no pot masperanca por venca

\numero{5} No puch mostrar lo secret de ma penssa

\numero{6} E vanament he por d\textit{e}la resposta

\numero{7} Lo meu duptar maior duptem acosta

\numero{8} Ffemenill gest ardiment me defenssa

\end{estrofa}



\begin{estrofa}

\numero{9} Alguns han dit que uergonya nos troba

\numero{10} Mas yom pusch fer daquella testimonj

\numero{11} De ujsta no semblant es al dimonj

\numero{12} Part de mos senys e parlar me derroba

\numero{13} Dona sentir de si alguns forts actes

\numero{14} Segons de molts hauem hoydes gestes

\numero{15} Crehent los tals quj descolen les festes

\numero{16} Senyor es meu amor fermals contractes

\end{estrofa}



\begin{estrofa}

\numero{17} E si posques lobedienca tolrre

\numero{18} Que per son colp la mjamor jau morta

\numero{19} Car no parent les altra mou estorta

\numero{20} Nosse qujn deu dampnat sant me fa colrre

\numero{21} Yo hagrasforc de metre tost en obra

\numero{22} Lo que no gos mostrar una parenca

\numero{23} Ab gran esforc vergonyam fa temenca

\numero{24} Si cardiment vn poch en mj no sobra

\end{estrofa}



\begin{estrofa}

\numero{25} [41v] En tots aquells on gran amor no penja

\numero{26} Son giny no pot de uergonya ne forca

\numero{27} A mj he pochs ason uoler nos forca

\numero{28} Nostre uoler nostres enemjchs uenja

\numero{29} Aquest uoler desig yamor sostenen

\numero{30} C\interlin{ausa} \sout{cerquen fi} daquests sou uos aquj sesguarden

\numero{31} Creure no pusch vostres sentiments tarden

\numero{32} En descobrir los mals que per uos uenen

\end{estrofa}



\begin{estrofa}

\numero{33} O follamor aquells dolor sostenen

\numero{34} Que cerquen fi \sout{en} ala \interlin{on} no pot esser

\numero{35} De llur treball nos mostra res en esser

\numero{36} E son aquests los quj de uos sentenen

\end{estrofa}
