\begin{estrofa}

\numero{1} C\textsc{o}ratje meu / a pendre, sfor\c{c}, molt tart

\numero{2} no piados / de tots los quit sostenen

\numero{3} la'rma y lo cos / a departir se venen

\numero{4} per tu, ser flach / lo cos de viure's fart

\numero{5} mos vlls no son / liberts a son, ofici

\numero{6} mon pas, es tolt, / ma lengua, nom profita

\numero{7} e de a\c{c}o / la vergonyas delita

\numero{8} com so plagat / de tan vergonyos vici

\end{estrofa}



\begin{estrofa}

\numero{9} Pahor me sent / gran suor me comen\c{c}a

\numero{10} surtint, mo\textit{n} cor / lo pits me cuyda rompre

\numero{11} nom trobesfor\c{c} / per vergonya corro\textit{m}pre

\numero{12} esser no pot / ma esperan\c{c}a, p'or ven\c{c}a

\numero{13} no puch mostrar / lo secret de ma pensa

\numero{14} e vanament / he p'or de la resposta

\numero{15} lo meu duptar / major dupte macosta

\numero{16} femenill gest / ardiment me defensa

\end{estrofa}



\begin{estrofa}

\numero{17} Alguns han dit / que vergonya nos troba

\numero{18} mas yo puch fer / d'aquella testimoni

\numero{19} de vista no / semblant es al dimoni

\numero{20} part de mos senys / e parlar, me derroba

\numero{21} [72r] dona, sentir / de si, alguns forts aectes\footnote{\textit{Per }actes.}

\numero{22} segons de molts / hauem hoydes gestes

\numero{23} crehent los tals / qui descolen, les festes

\numero{24} sen\textit{y}or, es meu / amor, fer mal's contractes

\end{estrofa}



\begin{estrofa}

\numero{25} E sil posques / lo bedien\c{c}a tolre

\numero{26} que per son colp / la mi'amor, jau morta

\numero{27} car no parent / l'es altra'amor estrota\footnote{\textit{Per }estorta.}

\numero{28} no sequin deu / damnat sanct, me fa colre

\numero{29} yo ag'ra sfor\c{c} / de metre tost, en obra

\numero{30} lo que no gos / mostrar vna, paren\c{c}a

\numero{31} ab gran esfor\c{c} / vergonyam fa temen\c{c}a

\numero{32} si qu'ardiment / vn poch, en mi no sobra

\end{estrofa}



\begin{estrofa}

\numero{33} En tots  aquells / on gran amor / no penja

\numero{34} son giny / no pot / de vergonya, ne for\c{c}a

\numero{35} ami, he pochs / a son voler nos for\c{c}a

\numero{36} lo meu voler / los meus, enemichs'venja

\numero{37} aquest voler / desig, y amor, sostenen

\numero{38} causa d'aquests / sou vos, aqui ses guarden

\numero{39} creure no puch / vostres sentiments, tarden

\numero{40} en descobrir / los mals que per vos venen

\end{estrofa}
