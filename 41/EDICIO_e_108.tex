\poema{(e 108) [41]}\\

\begin{rubrica}

CANT MORAL. VI. 

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} [141v] V\textsc{o}lg'ra ser nat / cent anys o pus atras

\numero{2} perque son cert / que's pijorat lo mon

\numero{3} ay tant poch val / quell mateix no sab hon

\numero{4} sa'ulesa gran / en pus baix loch baixas.

\numero{5} E veig vn cas / de raho apartat

\numero{6} que lo mal hom / fa casa prop la f\'{o}rca

\numero{7} el home just / fuig es met en cau\'{o}rca

\numero{8} per aquell mal / que james ha pensat.

\end{estrofa}



\begin{estrofa}

\numero{9} Foll es aquell / qui vol hauer vsat

\numero{10} de mal saber / hon sera ben entes,

\numero{11} imaginar / deu, que sera repres

\numero{12} e sens murmur / no deu ser comportat.

\numero{13} Car lo mal hom / deu ser cridat ab trompa

\numero{14} per \c{c}o que'mi / e altres no engan,

\numero{15} tant son aquells / qui per bons homens van

\numero{16} que difamats / cessaria llur pompa.

\end{estrofa}



\begin{estrofa}

\numero{17} Lo just voler / pensa com encorrompa

\numero{18} ficte voler / acolorat de be,

\numero{19} alla doncs es / guardada bona fe

\numero{20} com lo leig fet / acusador lo trompa.

\numero{21} Aquell es dit / misericordi\'{o}s

\numero{22} que guarda'ls bons / que nols sia tort fet

\numero{23} perdonador / al qui leig fet comet

\numero{24} d'infinits mals / dona ocasions.

\end{estrofa}



\begin{estrofa}

\numero{25} [142r] Fama tement / l'hom es dit virtuos

\numero{26} e \c{c}o no cau / en hom desuergonyit

\numero{27} doncs lo mal dir / no deu ser en oblit

\numero{28} puix que virtut / mostra l'hom vicios.

\numero{29} Al vici som / moguts naturalment

\numero{30} e costreny nos / por de la mala fama

\numero{31} beneyt aquell / quel besa boca brama

\numero{32} e diu lo mal / com ben es conexent.

\end{estrofa}



\begin{estrofa}

\numero{33} Es mal senyal / del qui es desplasent

\numero{34} es mor de'nuig / com mal dir s'acost\'{u}ma

\numero{35} e le's molt greu / sins ne hou dir la suma

\numero{36} d'algun leig fet / vehent si mal mirent.

\numero{37} Si com aquell / qui'n forca veu morir

\numero{38} son companyo / qui nol passa del mal

\numero{39} e diu que es / cosa descominal

\numero{40} que per tal fet / dega mort sostenir.

\end{estrofa}

\numero{41} Ia de present / delit no puch sentir

\numero{42} sino semblant / aquell del temps passat,

\numero{43} e com lo serch / e tal nol he trobat

\numero{44} ami mateix / e tot lo mon hair.
