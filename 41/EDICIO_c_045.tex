\begin{estrofa}

\numero{1} V\textsc{o}lgra ser nat / cent anys ho pus atras

\numero{2} per que son cert / que's pijorat lo mon

\numero{3} aytant poch val / que'll mateix no sab hon

\numero{4} sa'ulesa gran / en pus baix loch baixas.

\numero{5} e veig vn cas / de raho apartat

\numero{6} que lo mal hom / fa casa prop la forca

\numero{7} hel home just / fuig, es met, en cauorca,

\numero{8} per aquell mal / que james ha pensat.

\end{estrofa}



\begin{estrofa}

\numero{9} Foll es aquell / qui vol hauer vsat

\numero{10} de mal saber / hon sera ben entes

\numero{11} imiginer\footnote{\textit{Per} imaginar.} / deu, que sera repres.

\numero{12} e sens murmur / no deu ser comportat,

\numero{13} e lo mal hom / deu ser cridat, ab trompa,

\numero{14} per \c{c}o que mi / he altres no engan

\numero{15} tants son aquells / qui per bons homens van

\numero{16} que diffamats / cessaria lur pompa.

\end{estrofa}



\begin{estrofa}

\numero{17} [38v] Lo just voler / pensa com en corrompa

\numero{18} ficte voler / acolorat de be

\numero{19} e lladonchs es / guardada bona fe

\numero{20} com lo leig fet / acusador lo trompa.

\numero{21} aquells es dit / misericordios

\numero{22} que guardals bons / que nols sie tort fet

\numero{23} perdonador / al qui leig fet comet

\numero{24} d'infinits mals / dona ocasios.

\end{estrofa}



\begin{estrofa}

\numero{25} Fama tement / l'hom es dit virtuos

\numero{26} e \c{c}o no cau / en hom desuergonyit

\numero{27} donchs lo mal dir / nodeu ser en oblit

\numero{28} puis que virtut / mostra l'hom vicios:

\numero{29} al vici som / moguts naturalment

\numero{30} e costreny nos / p'hor\footnote{\textit{Per} paor.} dela mala fama

\numero{31} beneyt aquell / que'l be'sa boca brama

\numero{32} e diu lo mal / com ben'es conexent.

\end{estrofa}



\begin{estrofa}

\numero{33} Es mal senyal / de quil'es desplasent

\numero{34} es mor de'nuig / com mal dir s'acostuma

\numero{35} e l'es molt greu / sins ne hou dir la suma

\numero{36} d'algun leig fet / vehent si mal mirent.

\numero{37} si com aquell / qui'n forca veu morir

\numero{38} son companyo / qui nol passa de mal

\numero{39} e diu que es / cosa descominal

\numero{40} que per tal fet / dega sostenir.

\end{estrofa}
