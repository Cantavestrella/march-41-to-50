\begin{estrofa}

\numero{1} [40r] Volgra ser nat / cent anys ha pus atras

\numero{2} Perque son sert ques pigorat lo mon

\numero{3} Aytant poch vall quel mateix no sap hon

\numero{4} Sa vilessa gran en pus baix loch baixas

\numero{5} E ueig hun cas de raho apertat

\numero{6} Que lo mall hom fa casa prop la forca

\numero{7} E lome just fuig es met en cauorca

\numero{8} Per aquell mal q\textit{ue} iames ha pensat

\end{estrofa}



\begin{estrofa}

\numero{9} Ffoll es aquell q\textit{ui} voll auer vssat

\numero{10} De mall saber hon sera be entes

\numero{11} Himaginar deu que sera repres

\numero{12} E sens murmur / no deu ser conportat

\numero{13} E lo mal hom deu ser cridat ab trompa

\numero{14} Per\c{c}o que mj / ne laltre no engan

\numero{15} Tants son aquels / quj per bons home\textit{n}s van

\numero{16} Que diffamats cessaria lur pompa

\end{estrofa}



\begin{estrofa}

\numero{17} Lo iust voler / pen\c{c}a com enter\interlin{r}ompa

\numero{18} Fficta voler acolorat de be

\numero{19} E la donchs es gordada bona fe

\numero{20} Com lo leig fet acusador lo trompa

\numero{21} Aquell es dit mjsericordios

\numero{22} Quj guardals bons / q\textit{ue} no sia tort fet

\numero{23} Perdonador al quj leig fet comet

\numero{24} Dinfinjts mals / dona ocasios

\end{estrofa}



\begin{estrofa}

\numero{25} [40v] Ffama tement lom es dit virtuos

\numero{26} E aco no cau en hom desuergonyit

\numero{27} Donchs lo mal dir no deu ser en oblit

\numero{28} Puys q\textit{ue} virtut / mostren lom vicios

\numero{29} A ujsi som moguts naturalment

\numero{30} E costreny nos por / dela mala fama

\numero{31} Beneyt aquell / qujl be sa boca brama

\numero{32} E diu lo mall com ben es conaxent

\end{estrofa}



\begin{estrofa}

\numero{34} Es mor danuig com mal dir sa costuma

\numero{35} E les molt greu / sins hou lauar la suma

\numero{36} Dalgun leig fet vahent si malmjrent

\numero{37} Si com aquell q\textit{ui}n forca veu morir

\numero{38} Son compa\textit{n}yo q\textit{ui} nol passa de mal

\numero{39} E diu ques molt cosa descomjnall

\numero{40} Que per tal fet / dega mort sostenjr

\end{estrofa}
