\begin{estrofa}

\numero{1} Volgra ser nat cent anys opus atras

\numero{2} perque son cert ques pijorat lo mon

\numero{3} aytant poch val quell mateix no sab hon

\numero{4} saulesagran enpus baix loch baixas

\numero{5} Eveig vncas deraho apartat

\numero{6} quelomal hom facasaprop la forca

\numero{7} elome just fuig es met en cauorca

\numero{8} peraquell mal que james hapensat

\end{estrofa}



\begin{estrofa}

\numero{9} Ffoll es aquell quivol hauer vsat

\numero{10} [37v] de mal saber hon sera ben entes

\numero{11} imaginar deuque serarepres

\numero{12} esens murmur no deu ser comportat

\numero{13} E lo mal homdeu ser cridat ab trompa

\numero{14} per\c{c}o que mi e altre no engan

\numero{15} tants sonaquells qui per bonshomens van

\numero{16} que diffamats cessaria lurpompa

\end{estrofa}



\begin{estrofa}

\numero{17} Lo just voler pensacom encorrompa

\numero{18} fictevoler acolorat debe

\numero{19} eladonchs es guardada bonafe

\numero{20} com lo leig fet acusador lo trompa

\numero{21} Aquell es dit misericordios

\numero{22} queguardals bens que nolssiatort fet

\numero{23} perdonador  alqui leig fet comet

\numero{24} dinfinits mals dona occasios

\end{estrofa}



\begin{estrofa}

\numero{25} Ffamatement lom es dit virtuos

\numero{26} ea\c{c}o no cau en homdesuergonyit

\numero{27} donchs lo maldir no deu ser enoblit

\numero{28} puis quevirtut mostra lom vicios

\numero{29} A vici som moguts naturalment

\numero{30} ecostreny nos por delamala fama

\numero{31} beneyt aquell quelbe sa boca brama

\numero{32} ediu lomal com ben es conexent

\end{estrofa}



\begin{estrofa}

\numero{33} Es mal senyal del quil es desplasent

\numero{34} es mor denuig com mal dir sacostuma

\numero{35} eles molt greu fins \interlin{ne} hou dir la suma

\numero{36} dalgun leig fet vehent si mal mirent

\numero{37} [38r] Sicom aquell quin forca veu morir

\numero{38} son companyo qui nol pasa de mal

\numero{39} ediu ques cosa molt descominal

\numero{40} quepertal fet dega mort sostenir

\end{estrofa}



\begin{estrofa}

\numero{41} Jade present delit no puch sentir

\numero{42} sino semblant aquell del tempspassat

\numero{43} ecomlo cerch etal nol he trobat

\numero{44} ami mateix e tot lo mon ahir

\end{estrofa}
