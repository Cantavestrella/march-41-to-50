\begin{estrofa}

\numero{1} V\textsc{o}lgra: ser nat / cent, anys, ho pus, atras

\numero{2} per que son cert / que's pijorat lo mon

\numero{3} aytant poch val / que'll mateix, no sab, hon

\numero{4} sa'ulesa \cite{ref36} gran / en pus baix loch, baixas

\numero{5} he veig vn cas / de raho, apartat

\numero{6} que lo mal hom / fa casa prop, la forca

\numero{7} hel home just / fuig, es met, encauorca \cite{ref37}

\numero{8} per, aquell mal / que james ha pensat

\end{estrofa}



\begin{estrofa}

\numero{9} Foll es aquell / qui vol haver, vsat

\numero{10} de mal saber / hon sera, ben, entes

\numero{11} imaginar / deu, que sera repres

\numero{12} he sens murmur / no deu ser comportat

\numero{13} he lo mal hom / deu ser cridat, ab trompa

\numero{14} per \c{c}o que mi / he altres no, engan

\numero{15} tants son, aquells / qui per bo\textit{n}s, homens, van

\numero{16} que diffamats / cessaria lur, pompa

\end{estrofa}



\begin{estrofa}

\numero{17} Lo, iust voler / pensa, com, encorrompa

\numero{18} ficte voler / a colorat: de be

\numero{19} he lla donchs es / guardada, bona, fe

\numero{20} com lo leig fet / acusador lo, trompa

\numero{21} [36v] aquell es dit / misericordios

\numero{22} que guarda'ls bons / que nols sie, tort, fet

\numero{23} perdonador / al qui leig fet, comet

\numero{24} d'infinits mals / dona, occasios

\end{estrofa}



\begin{estrofa}

\numero{25} Fama tement / l'hom, es dit, virtuos

\numero{26} he'\c{c}o, no cau / en hom, desuergonyit

\numero{27} donchs, lo mal dir / nodeu ser, en oblit

\numero{28} puis que virtut / mostra l'hom vicios

\numero{29} al vici, som / moguts, naturalment

\numero{30} he costreny nos / p'hor, de la mala fama

\numero{31} beneyt, aquell / que'l be, sa boca, brama

\numero{32} he, diu lo mal / com be n'es, conexent

\end{estrofa}



\begin{estrofa}

\numero{33} Es mal senyal / del quil'es desplasent

\numero{34} es mor, de'nuig / com mal dir, s'acostuma

\numero{35} he l'es molt greu / sins ne hou dir, la suma

\numero{36} d'algun leig fet / vehent, si, mal mirent

\numero{37} sicom aquell / qui'n forca veu morir

\numero{38} son companyo / qui nol passa, de mal

\numero{39} he diu, que es / cosa descominal

\numero{40} que per tal fet / dega mort sostenir

\end{estrofa}
