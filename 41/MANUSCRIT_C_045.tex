\begin{estrofa}

\numero{1} Volgra ser nat / cent anys ho pus atras

\numero{2} perque son cert / qu'es pijorat lo mon

\numero{3} aytant poch ual / qu'el meteix no sap hon

\numero{4} sa'ulesa gran / en pus baix loch baixas

\numero{5} [56v] e ueig un cas / de raho apartat

\numero{6} quelo mal hon / fa casa prop la forca

\numero{7} hel home iust / fuig es'met en cauorca

\numero{8} per aquell mal / que iames ha pensat.

\end{estrofa}



\begin{estrofa}

\numero{9} Foll es aquell / qui uol hauer usat

\numero{10} de mal saber / hon sera ben entes

\numero{11} jmaginar deu / que sara repres

\numero{12} esens murmur / no deu ser comportat

\numero{13} elo mal hom / deu ser cridat ab trompa

\numero{14} per\c{c}o que mi / e altres no engan

\numero{15} tans son aquells / qui per bons homens uan

\numero{16} que diffamats / cessaria lur pompa.

\end{estrofa}



\begin{estrofa}

\numero{17} Lo just uoler / pensa com en corrompa

\numero{18} ficte uoler / ab colorat de be

\numero{19} ella donchs es / guardada bona fe

\numero{20} com lo leig fet / accusador lo trompa

\numero{21} aquells es dit / misericordios

\numero{22} que guardals bons / que nols sia tort fet

\numero{23} perdonador / alqui leig fet comet

\numero{24} d'infinits mals / dona occasiosis

\end{estrofa}



\begin{estrofa}

\numero{25} [57r] Fama tement / l'hom es dit uirtuos

\numero{26} e \c{c}o no cau / en hom desuergonyit

\numero{27} donchs lo mal dir / no deu ser ennobilit

\numero{28} puis que uirtut / mostra l'hom ui\c{c}ios.

\numero{29} al ui\c{c}i som / moguts naturalment

\numero{30} e constreny nos / p'hor de mala fama

\numero{31} beneyt aquell / quel be sa boca brama

\numero{32} e diu lo mal / com ben es coneixent.

\end{estrofa}



\begin{estrofa}

\numero{33} Es mal senyal / del quil'es desplasent

\numero{34} es'mor d'enuig / com mal dir s'acostuma

\numero{35} el'nes molt greu / sius ne hou dir la suma

\numero{36} d'algun leig fet / uehent si mal mirent.

\numero{37} si com aquell / qu'en forca ueu morir

\numero{38} son companyo / que nol passa de mal

\numero{39} e diu que es / cosa descominal

\numero{40} que per tal fet / dega mort sostenir

\end{estrofa}
