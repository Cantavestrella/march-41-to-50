\poema{(E 120) [41]}\\

\begin{rubrica}

Moss\textit{en }ausias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Volgra ser nat / cent anys o pus atras

\numero{2} per que so cert / ques pijorat lo mon

\numero{3} aytant poch val / quell mateix no sab hon

\numero{4} saulesa gran / en pus baix lloch baixas

\numero{5} E veig vn cas / de raho apartat

\numero{6} que lo mal hom / fa casa prop la forca

\numero{7} e lome just / fuig hes met en cauorca

\numero{8} per aquell mal / que James ha pensat

\end{estrofa}



\begin{estrofa}

\numero{9} [184v] Foll es aquell / qui vol hauer usat

\numero{10} de mal saber / hon sera ben entes

\numero{11} ymaginar / deu que sera repres

\numero{12} e sens mur mur / no deu ser comportat

\numero{13} Car lo mal hom / deu ser cridat ab trompa

\numero{14} per \c{c}o que mi / e laltre no engan

\numero{15} tants son aquells / qui per bons homens van

\numero{16} que diffamats / cessaria llur pompa

\end{estrofa}



\begin{estrofa}

\numero{17} Lo just voler / pensa com entorrompa

\numero{18} ficte voler / acolorat de be

\numero{19} e lla donchs es / guardada bona fe

\numero{20} com lo lleig fet / acusador lo trompa

\numero{21} Aquell es dit / misericordios

\numero{22} qui guardals bons / que nols sia tort fet

\numero{23} perdonador / al qui lleig fet comet

\numero{24} dinfinits mals / dona ocasios

\end{estrofa}



\begin{estrofa}

\numero{25} Fama tement / lom es dit virtuos

\numero{26} e \c{c}o no cau / en lom desuergonyit

\numero{27} donchs lo mal dir / no deu ser en oblit

\numero{28} puis que virtut / mostra lom vi\c{c}ios

\numero{29} Al vi\c{c}i som / moguts naturalment

\numero{30} e costreny nos / por dela mala fama

\numero{31} beneyt aquell / quil be sa boca brama

\numero{32} e diu lo mal / com be nes conexent

\end{estrofa}



\begin{estrofa}

\numero{33} [185r] Es mal senyal / del qui les desplasent

\numero{34} e mor denuig / com al dir sa costuma

\numero{35} e les molt greu / sius ou lleuar la suma

\numero{36} dalgun lleig fet / vehent si malmirent

\numero{37} Si com aquell / quin forca veu morir

\numero{38} son companyo / qui nol passa de mal

\numero{39} e diu ques molt / cosa descominal

\numero{40} que per tal fet / deia mort sostenir

\end{estrofa}

\numero{42} sino semblant / aquell del temps passat

\numero{44} ami mateix / e tot lo mon ahyr
