\begin{estrofa}

\numero{1} \textparagraph\  Volgraser nat cent ayns ho pus atras

\numero{2} Perque son cert ques pijorat lomon

\numero{3} Aytanpoch val quell matex no sab hon

\numero{4} Saulessa gran enpus bax loch baxas

\numero{5} Eueig vn cas de raho apartat

\numero{6} Quelo mal hom fa cassa prop la forca

\numero{7} E lome just fug es met encauorqua

\numero{8} Per aquell mal que yames hapensat

\end{estrofa}



\begin{estrofa}

\numero{9} [81] Foll es aquell qui uol hauer vssat

\numero{10} Demal saber on sera ben entes

\numero{11} Ymaginar deu que sera repres

\numero{12} E sent murmur no deu ser comportat

\numero{13} Elomal hom deu ser cridat ab trompa

\numero{14} Perco que mj elaltre no engan

\numero{15} Tants son aquell quj per bons homens van

\numero{16} Que difamats resaria lur pompa

\end{estrofa}



\begin{estrofa}

\numero{17} Lo just valer pensa com enterompa

\numero{18} Ficte voler acolorat debe

\numero{19} Ela donchs es guardada bona fe

\numero{20} Com lo leg fet acusador lo trompa

\numero{21} Aquell es da misericordios

\numero{22} Quj guardals bons que nols sia tort fet

\numero{23} Perdonador aquj leg fet comet

\numero{24} Dinfinits mal dona ocasios

\end{estrofa}



\begin{estrofa}

\numero{25} Ffama tement lom es dit virtuos

\numero{26} Aco no cau enhom desuergoyat

\numero{27} Donchs lomal dir no deu ser enoblit

\numero{28} Pusque virtut mostran lom vicios

\numero{29} Ha vicis som moguts naturalment

\numero{30} E costreny nos por dela mala fama

\numero{31} Beneyt aquell quil be saboca brama

\numero{32} E diu lo mal combenes conexent

\end{estrofa}



\begin{estrofa}

\numero{33} [82] Es mal senyal del quj les desplasent

\numero{34} E mor denug com mal dir sacostuma

\numero{35} Elj desplau si\textit{us} hou leuar lasuma

\numero{36} Dalgun leg fet vehent si malmjrent

\numero{37} Sicom aquell quin forca veu morir

\numero{38} Son companyo quj nol passa demal

\numero{39} E diuques molt cossa descomjnal

\numero{40} Que per tal fet dega mort sofferir

\end{estrofa}
