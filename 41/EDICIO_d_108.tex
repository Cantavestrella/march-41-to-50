\begin{estrofa}

\numero{1} V\textsc{o}lg'ra ser nat / \c{c}ent anys o pus atras

\numero{2} per que son cert / que's pijorat lo mon

\numero{3} ay tant poch val / q\textit{ue}ll mateix no sab hon

\numero{4} sa'ulesa gran / en pus baiz lloch baixas.

\numero{5} E veig vn cas / de raho apartat

\numero{6} que lo mal hom / fa casa prop la f\'{o}rca

\numero{7} el home just / fuig es met en cau\'{o}rca

\numero{8} per aquell mal / que james ha pensat.

\end{estrofa}



\begin{estrofa}

\numero{9} \textparagraph\  Foll es aquell / qui vol hauer vsat

\numero{10} de mal saber / hon sera ben entes,

\numero{11} imiginar / deu,que sera repres.

\numero{12} e sens murmur / no deu ser comportat,

\numero{13} Car lo mal hom / deu ser cridat ab trompa

\numero{14} per \c{c}o que'mi / e altres no engan,

\numero{15} tants son aquells / qui per bons homens van

\numero{16} que diffamats / \c{c}essaria llur pompa.

\end{estrofa}



\begin{estrofa}

\numero{17} \textparagraph\  Lo just voler / pensa com encorrompa

\numero{18} ficte voler / acolorat de be,

\numero{19} alla donchs es / guardada bona fe

\numero{20} com lo leig fet / acusador lo trompa.

\numero{21} Aquells es dit / missericordi\'{o}s

\numero{22} que guarda'ls bons / que nols sia tort fet

\numero{23} perdonador / alqui leig fet comet

\numero{24} d'infinits mals / dona ocasi\'{o}s.

\end{estrofa}



\begin{estrofa}

\numero{25} [148v] \textparagraph\  Fama tement / l'hom es dit virtuos

\numero{26} e\c{c}o no cau / en hom desuergonyit,

\numero{27} donchs lo mal dir / no deu ser en oblit

\numero{28} puis que virtut / mostra le\footnote{\textit{Per} lo.} hom vicios.

\numero{29} Al vici som / moguts naturalment

\numero{30} e costreny nos / por dela mala fama

\numero{31} beneyt aquell / quel be sa boca brama

\numero{32} e diu lo mal / com ben es conexent.

\end{estrofa}



\begin{estrofa}

\numero{33} \textparagraph\  Es mal senyal / del qui es desplasent

\numero{34} es mor d'nuig / com mal dir s'acost\'{u}ma

\numero{35} ele's molt greu / sins ne hou dir la suma

\numero{36} d'algun leig fet / vehent si mal mirent.

\numero{37} Si com aquell / qui'n forca veu morir

\numero{38} son companyo / qui nol passa del mal

\numero{39} e diu que's / cosa descominal

\numero{40} que per tal fet / dega mort sostenir.

\end{estrofa}

\numero{41} \textparagraph\  Ia de present / delit no puch sentir

\numero{42} sino semblant / aquell del temps passat,

\numero{43} e com lo \c{c}erch / e tal nol he trobat

\numero{44} ami mateix / e tot lo mon hair.
