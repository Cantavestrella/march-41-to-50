\begin{estrofa}

\numero{1} v Olgrasser nat cent anys o pus atras

\numero{2} [40r] Per que som cert ques pijorat lo mon

\numero{3} Aytan poch val q\textit{ue}ll mat\textit{e}x no sab hon

\numero{4} Saulesa gran en pus bax loch baxas

\numero{5} E vetx vn cas de raho apartat

\numero{6} Que lo mal hom fa casa prop  la forq\textit{ua}

\numero{7} E lome just futx es met en cauorqua

\numero{8} Per aq\textit{ue}ll mal q\textit{ue} ya mes ha pensat

\end{estrofa}



\begin{estrofa}

\numero{9} Foll es aq\textit{ue}ll qui vol hauer vsat

\numero{10} De mal saber hon sera ben antes

\numero{11} Ymaginar deu q\textit{ue} sera repres

\numero{12} E sens murmur no deu ser co\textit{m}portat

\numero{13} E lo mal hom deu ser cridat ab tro\textit{m}pa

\numero{14} Per\c{c}o q\textit{ue} mj e laltre no engan

\numero{15} Tant son aq\textit{ue}lls qui p\textit{er} bo\textit{n}s home\textit{n}s van

\numero{16} Que difamats \c{c}essaua llur pompa

\end{estrofa}



\begin{estrofa}

\numero{17} Lo iust voler penssa com entorrompa

\numero{18} Ficte voler acolorat de be

\numero{19} E lladonchs es guardada bona fe

\numero{20} Com lo letx fet acusador lo tro\textit{m}pa

\numero{21} Aquell es dit misericordios

\numero{22} Qui guardals bons q\textit{ue} nols sia tort fet

\numero{23} Perdonador al qui letx fet comet

\numero{24} Dinfinjts mals dona occasions

\end{estrofa}



\begin{estrofa}

\numero{25} Fama tement lom es dit virtuos

\numero{26} [40v] E so no cau en hom desuergonyit

\numero{27} Donchs lo mal dir no deu ser en oblit

\numero{28} Puysque virtut mostran lom vi\c{c}ios

\numero{29} A ui\c{c}i som moguts naturalment

\numero{30} E constrenynos por dela mala fama

\numero{31} Beneyt aq\textit{ue}ll quil be sa bocha brama

\numero{32} E diu lo mal com be nes conexent

\end{estrofa}



\begin{estrofa}

\numero{33} Es mal senyal del quil es desplaent

\numero{34} E mor danutx com mal dir sacostuma

\numero{35} E li desplau sius hou leuar la suma

\numero{36} Dalgun letx fet vehent si mal mjrent

\numero{37} Si co\textit{m} aquell quin forca veu morir

\numero{38} Son co\textit{m}panyo qui nol passa de mal

\numero{39} E diu ques molt cosa descomjnal

\numero{40} Que per al fet deia mort sostenir

\end{estrofa}
