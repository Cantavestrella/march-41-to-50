\poema{(c 69) [42]}\\

\begin{estrofa}

\numero{1} V\textsc{o}s qui sabeu / dela Tortra costum

\numero{2} e si nou feu / placi al vos hoyr.

\numero{3} [66v] quant mort litol / son par se vol jaquir

\numero{4} d'obres d'amor / no beu aygua de flum.

\numero{5} ans en los clots / ensutza primer layga

\numero{6} nos posa may / en vert arbre fullat

\numero{7} mas contra'\c{c}o / es vostra qualitat

\numero{8} per gran desig / no cast, quem vos se'rayga.

\end{estrofa}



\begin{estrofa}

\numero{9} E no cuydeu / dona que bens es caiga

\numero{10} que puis hagues / tastat la carn gentil

\numero{11} a mercader / liuras vostre cos vil

\numero{12} e son dret nom / en Ioan me pens caiga.

\numero{13} e si voleu / queus ne don coneixen\c{c}a

\numero{14} la fa\c{c} es gran / ab la vista molt losca

\numero{15} sos fonaments / son de lagost e mosca

\numero{16} cert no mereix / drap vendre de floren\c{c}a.

\end{estrofa}



\begin{estrofa}

\numero{17} E coneixent / la vostra grans fallen\c{c}a

\numero{18} volgues muntar / en amar Caualler

\numero{19} e sabent ell / tot vostre fet en ver

\numero{20} en vos amar / se tengra concien\c{c}a.

\numero{21} sabent molt clar / la ensutzada vida

\numero{22} prenint publich / les pagues del peccat

\numero{23} vostre cos leig / per drap es baratat

\numero{24} vostre seruir / es bo sols per a dida.

\end{estrofa}



\begin{estrofa}

\numero{25} E no cuydeu / fillaus hagues jaquida

\numero{26} vos alletant / aquella'n vostre llet

\numero{27} car vostre cos / es de veri replet

\numero{28} e mostren ho / vostres'pels, foramida:

\numero{29} car sius jaquiu / vostra barba criada

\numero{30} [67r] e laus toleu / puis ab los pels dels bra\c{c}as

\numero{31} poran sen fer / auant at josos la\c{c}os

\numero{32} prenint Perdiu / ho Tortra, o Bequada.

\end{estrofa}



\begin{estrofa}

\numero{33} Quant oyreu / alcauota prouada

\numero{34} responeu tost / que per vos ho diran

\numero{35} e puis per nom / propri, vos cridaran

\numero{36} ja nous mostreu / en l'oir empachada.

\numero{37} enterrogant / amichs e que voleu

\numero{38} endret d'amor / hay res que yo fer pusca

\numero{39} tracte semblant / james me troba cusca

\numero{40} presta sere / a quant demanareu.

\end{estrofa}
