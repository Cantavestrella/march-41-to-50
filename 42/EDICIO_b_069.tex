\begin{estrofa}

\numero{1} [63v] V\textsc{o}s qui sabeu / dela Tortral'costum

\numero{2} e si nou feu / placi'al vos oyr

\numero{3} quant mort li tol / son par, se vol jaquir

\numero{4} d'obres d'amor / no beu aygua de flum

\numero{5} ans en los clots / ensutza primer layga

\numero{6} nos posa may / en vert, arbre fullat

\numero{7} mas contra \c{c}o / es vostra qualitat

\numero{8} per gra\textit{n} desig / no cast que\textit{n} vos, se'rayga

\end{estrofa}



\begin{estrofa}

\numero{9} E no cuydeu / dona que beus escaiga

\numero{10} que puis hagues / tastat la carn gentil

\numero{11} a mercader / liuras vostre cors vil

\numero{12} e son dret nom / en Ioan, me pens, caiga

\numero{13} e si voleu / queus ne don coneixen\c{c}a

\numero{14} sa fa\c{c} es gran / abla vista molt, los\c{c}a

\numero{15} sos fonaments / son de lagost, e mos\c{c}a

\numero{16} cert no mereix / draps vendre, de floren\c{c}a

\end{estrofa}



\begin{estrofa}

\numero{17} E coneixent / la vostra gran fallen\c{c}a

\numero{18} volgues muntar / en amar Caualler

\numero{19} e sa bent ell / tot vostre fet, en ver

\numero{20} en vos amar / se tengra concien\c{c}a

\numero{21} sabent molt clar / la ensutzada vida

\numero{22} prenint publich / les pagues, del peccat

\numero{23} vostre \c{c}os leig / per drap, es baratat

\numero{24} vostre seruir / es bo, sols pera dida

\end{estrofa}



\begin{estrofa}

\numero{25} E no cuydeu / fillaus hagues jaquida

\numero{26} vos alletant / aquella'n vostra llet

\numero{27} car\c{c}\footnote{\textit{Per }car.} vostre cos / es de veri, replet

\numero{28} e mostren ho / vostres pels, fora mida

\numero{29} [64r] car sius jaquiu / vostra barba, criada

\numero{30} e laus toleu / puis ab los pels, dels bra\c{c}as\footnote{\textit{Per
}bra\c{c}os.}

\numero{31} poran sen fer / auantatjosos la\c{c}os

\numero{32} prenint perdiu / ho tortra, o bequada

\end{estrofa}



\begin{estrofa}

\numero{33} Quant oyreu / alcavota prouada

\numero{34} responeu tost / que per vos, ho diran

\numero{35} e puis per nom / propri, vos cridaran

\numero{36} ja nous mostreu / en lo'ir empachada

\numero{37} enterrogant / amichs, e que voleu

\numero{38} endret d'amor / hay res, que yo, fer pusca

\numero{39} tracte semblant / james, me troba, cusca \cite{ref44}

\numero{40} presta sere / a quant demanareu

\end{estrofa}
