\poema{(I 62) [42]}\\

\begin{rubrica}

Mossen Auzias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Os qui sabeu / dela \sout{coraltr} tortral costum

\numero{2} Esi nol feu / placiel vos hoyr

\numero{3} Quant mort li tol / son par si vol jaquir

\numero{4} Dobres damor / ne beu aygua de flum

\numero{5} Ans en los clots / ensutza primer laygua

\numero{6} Nes posa may / en lo arbre fullat

\numero{7} Mas contra\c{c}o / es vostra calitat

\numero{8} Per gran voler / no cast quen vos seraygua

\end{estrofa}



\begin{estrofa}

\numero{9} E no penseu / dona que be\textit{us} escaygua

\numero{10} Que pus hagues / tastat la carn gentil

\numero{11} Amercader / liuras vostre cors vil

\numero{12} E son dret nom / en Johan \sout{pens} me pens caygua

\numero{13} E si voleu / que\textit{us} ne do conexenca

\numero{14} Sa fas es gran / ab la vista molt loscha

\numero{15} Sos fonaments / son de legost o moscha

\numero{16} Cert no mereix / vendre drap de floren\c{c}a

\end{estrofa}



\begin{estrofa}

\numero{17} Vos conaxeu / la vostra gran fallenca

\numero{18} Volgues muntar / en amar caualler

\numero{19} Esabent ell / tot vostre \sout{pler en fet} fet en pler

\numero{20} De vos amar / se dara conscien\c{c}a

\numero{21} Sabent molt clar / molt sutzeada vida

\numero{22} Prenent publich / les plagues del peccat

\numero{23} Vostre cors leig / per drap es baratat

\numero{24} Vostre seruey / es sol bo per adida

\end{estrofa}



\begin{estrofa}

\numero{25} [76v] E no cuydeu / fille\textit{us} hagues jaquita

\numero{26} Vos alletant / aquell ab vostre let

\numero{27} Car vostre cors / es de veri replet

\numero{28} Emostren ho / vostres pels fora mida

\numero{29} Car si\textit{us} jaquiu / vostra barba criada

\numero{30} E la\textit{us} toleu / puys ab los pels dels brassos

\numero{31} Poran sen fer / auentatgosos la\c{c}os

\numero{32} Prenjnts perdi\textit{us} / o tortres sens tardada

\end{estrofa}



\begin{estrofa}

\numero{33} Quant hoyreu / alcauota prouada

\numero{34} Responeu prest / que per vos ho diran

\numero{35} Epuys per nom / propi\textit{us} nomenaran

\numero{36} En lo respost / no\textit{us} mostreu empatxada

\numero{37} Interrogant / amichs e que voleu

\numero{38} En fets damor / ay res que jo fer pusque

\numero{39} En tracte tal / iames me trobi cruscha

\numero{40} Presta sere / aquant demanareu

\end{estrofa}
