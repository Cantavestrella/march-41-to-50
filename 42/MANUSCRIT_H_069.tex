\poema{(H 69) [42]}\\

\begin{rubrica}

\pagina{[85v]} Mos\textit{sen} auzias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Uos qui sabeu / dela tortrall costum

\numero{2} E si nou feu placial vos hoir

\numero{3} Quant mort li toll / son par se vol iaq\textit{ui}r

\numero{4} Dobres damor / ne beu aygua d\textit{e} flum

\numero{5} Ans en los clots ensutza p\textit{ri}mer laygua

\numero{6} Nes posa may en vert arbre fullat

\numero{7} Mas contra\c{c}o es vostra calitat

\numero{8} Per gran desig / no cast quen vos sa raygua

\end{estrofa}



\begin{estrofa}

\numero{9} E no cuydeu dona que beus escaygua

\numero{10} Que puys hagues tastat la carn gentill

\numero{11} A mercader lliuras vostre cors vil

\numero{12} E son dret nom en iohan me pens caygua

\numero{13} E si uoleu queus nedo conexen\c{c}a

\numero{14} Sa fa\c{c} es gran ab la vista molt losca

\numero{15} Sos fonaments son d\textit{e} lagost o mosca

\numero{16} Cert no mereix draps vendre d\textit{e} floren\c{c}a

\end{estrofa}



\begin{estrofa}

\numero{17} E conaxent la vostra gran falen\c{c}a

\numero{18} Volgues muntar en amar caualer

\numero{19} E sabent ell tot vostre fet en ver

\numero{20} En uos amar / se te aconsien\c{c}a

\numero{21} Sabent molt clar lansutzeada vida

\numero{22} Prenent pubblich / les pagues dels peccat

\numero{23} Vostre cors leig / per drap es baretat

\numero{24} Vostre serujr es bo sol per ha dida

\end{estrofa}



\begin{estrofa}

\numero{25} [86r] E no cuydeu fillaus hagues iaq\textit{ui}da

\numero{26} Vos alletant aquel ab vostra llet

\numero{27} Car vostre cors es d\textit{e} ueri replet

\numero{28} E mostrenho vostres pels fora mjda

\numero{29} Car sius jaquju vostra barba criada

\numero{30} E laus toleu puys ab los pels d\textit{e}ls bra\c{c}os

\numero{31} Poran sen fer auentegossos la\c{c}os

\numero{32} Prenjnts predius / o tortra o bequada

\end{estrofa}



\begin{estrofa}

\numero{33} Quant hoyreu alcauota prouada

\numero{34} Responeu tost q\textit{ue} per vos ho diran

\numero{35} E puys per nom propi vos cridaran

\numero{36} Ja nous mostreu en loir enpatxada

\numero{37} Enterrogant amjchs e q\textit{ue} voleu

\numero{38} En dret damor voleu res q\textit{ue} fer pusque

\numero{39} Tracte semblant iames me troba cusqua

\numero{40} Presta sere / aquant demanareu

\end{estrofa}
