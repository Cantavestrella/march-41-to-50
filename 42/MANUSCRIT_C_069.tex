\begin{estrofa}

\numero{1} Vos qui sabeu / dela tortra costum

\numero{2} esi nou feu / pla\c{c}i aluos hoyr

\numero{3} quant mort li tol / son par se uol jaquir

\numero{4} d'obres d'amor / nobeu aygua de flum

\numero{5} ans e\textit{n}los clots / ensutza primer l'ayga

\numero{6} nos posa may / en uert arbre fullat

\numero{7} mas contra \c{c}o / es uostra qualitat

\numero{8} pergran desig / no cast quen uos serayga

\end{estrofa}



\begin{estrofa}

\numero{9} Eno cuydeu / dona que bens escayga

\numero{10} que puis hagues / tastat la carn gentil

\numero{11} amercader / liuras uostre cos uil

\numero{12} e son dret nom / es Ioan me pens cayga

\numero{13} e si uoleu / queus ne don coneixen\c{c}a

\numero{14} sa fa\c{c} es gran / ab la uista molt losca

\numero{15} sos fonaments / son del agost e mosca

\numero{16} cert no mereix / drap uendre de floren\c{c}a

\end{estrofa}



\begin{estrofa}

\numero{17} [94r] E coneixent / la uostra grans fallen\c{c}a

\numero{18} uolgues muntar / en amar caualler

\numero{19} e sabent ell / tot uostre fet en uer

\numero{20} en uos amar / se tengra con\c{c}ien\c{c}a

\numero{21} sabent mol clar / la ensutzada uida

\numero{22} prenint publich / les pagues del peccat

\numero{23} uostre cos leig / per drap esbaratat

\numero{24} uostre seruir / esbo sols per adida

\end{estrofa}



\begin{estrofa}

\numero{25} Eno cuydeu / filla us haues jaquida

\numero{26} uos alletant / aquella'n uostre llet

\numero{27} car uostre cos / es de ueri replet

\numero{28} e mostren ho / uostres pels foramida

\numero{29} car sius jaquiu / uostra barba criada

\numero{30} e laus toleu / puis ablos pels dels bra\c{c}os

\numero{31} poran ser fer / auant atiosos la\c{c}os

\numero{32} preuint Perdiu / ho Tortra o Bequada

\end{estrofa}



\begin{estrofa}

\numero{33} Quant oyreu / alcauota prouada

\numero{34} responeu tost / que peruos ho diran

\numero{35} e puis per nom / propri us cridaran

\numero{36} ya nous mostreu / en l'oyr empachada

\numero{37} [94v] enterroga\textit{n}t / amichs eque uoleu

\numero{38} endret d'amor / hay res que yo fer pusca

\numero{39} tacte semblant / james me troba cusca

\numero{40} presta sere / a quant demanareu

\end{estrofa}
