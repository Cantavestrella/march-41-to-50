\begin{estrofa}

\numero{1} Vos quj sabeu dela tortral costum

\numero{2} e si nou feu placial vos hoyr

\numero{3} quant mort li tol son par se vol jaqujr

\numero{4} dobres damor ne beu aygua de flum

\numero{5} ans en los clots ensutza primer laygua

\numero{6} nes posa may en vert arbre fullat

\numero{7} mas contra a\c{c}o es vostra qualitat

\numero{8} per gran desig no cast quan vos se raygua

\end{estrofa}



\begin{estrofa}

\numero{9} [45v] E no cuydeu dona que beus escaygua

\numero{10} que puys hagues tatast la carn gentil

\numero{11} a mercader liuras vostre cors vil

\numero{12} eson dret nom en Johan me pens caygua

\numero{13} e si voleu queus ne don conexen\c{c}a

\numero{14} sa faz es gran ab la vista molt llosqua

\numero{15} sos fonaments son de llagost o mosqua

\numero{16} cert no merex vendre draps de floren\c{c}a

\end{estrofa}



\begin{estrofa}

\numero{17} E conexent la vostra grant fallen\c{c}a

\numero{18} volgues muntar en amar caualler

\numero{19} e sabent ell tot vostre fet en ver

\numero{20} en vos amar se tengra concien\c{c}a

\numero{21} sabent molt cert lansutzada vida

\numero{22} prenent publich les pagues del peccat

\numero{23} vostre cos leig per drap es baratat

\numero{24} vostre serujr es bo sol pera dida

\end{estrofa}



\begin{estrofa}

\numero{25} No cuydeu fillaus hagues jaquida

\numero{26} vos alletant aquellab vostra llet

\numero{27} car vostre cos es de veri replet

\numero{28} e mostren ho vostres pels fora mjda

\numero{29} car sius jaquiu vostra barba criada

\numero{30} e laus toleu puix ab los pels dels bra\c{c}os

\numero{31} poran sen fer auantaiosos lla\c{c}os

\numero{32} prenjnts perdius etortra o bequada

\end{estrofa}



\begin{estrofa}

\numero{33} [46r] Quan hoyreu alcauota prouada

\numero{34} responeu tost que per vos ho diran

\numero{35} e puix per nom propi vos cridaran

\numero{36} jaus nos mostreu en loyr empachada

\numero{37} enterroguant amichs eque voleu

\numero{38} en dret damor hay res que yo fer pusca

\numero{39} tracte semblant james me trobe cusca

\numero{40} presta sere aquant demanareu

\end{estrofa}
