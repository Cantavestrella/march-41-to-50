\begin{rubrica}

DE AMOR CANT. LX.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} V\textsc{o}s que sabeu / de la Tortr'al cost\'{u}m

\numero{2} e si nou feu / placial vos ho\'{\i}r

\numero{3} quant mort li tol / son par se vol jaquir

\numero{4} d'obres d'Amor / no beu aygua de fl\'{u}m.

\numero{5} Ans en los clots / ensutza primer la'ygua

\numero{6} nos posa may / en vert arbre fullat.

\numero{7} mas contr'a\c{c}o / es vostra calit\'{a}t

\numero{8} per gran desig / no cast, qu'en vos se rayga.

\end{estrofa}



\begin{estrofa}

\numero{9} E no cuydeu / dona que beus escayga

\numero{10} que puix hagues / tastat la carn gentil

\numero{11} a mercader / liur\'{a}s vostre cos vil

\numero{12} \'{e} son dret nom / en Ioan me pens cayga.

\numero{13} E si voleu / queus ne don coneixen\c{c}a

\numero{14} sa faz es gran / ab la vista molt losca

\numero{15} sos fonaments / son de lagost e mosca

\numero{16} cert no mereix / drap vendre de Floren\c{c}a.

\end{estrofa}



\begin{estrofa}

\numero{17} [69r] E coneixent / la vostra gran fallen\c{c}a

\numero{18} volgues muntar / en armar caualler

\numero{19} e sabent ell / tot vostre fet en ver

\numero{20} de vos amar / se fara conscien\c{c}a.

\numero{21} Sabent molt clar / la ensutzada vida

\numero{22} prenint publich / les pagues del peccat

\numero{23} vostre cos leig / per drap es barat\'{a}t

\numero{24} vostre seru\'{\i}r / es bo sols pera d\'{\i}da.

\end{estrofa}



\begin{estrofa}

\numero{25} E no cuydeu / filla'us hagues jaqu\'{\i}da

\numero{26} vos alletant / aquella ab vostra let

\numero{27} car vostre cos / es de veri replet

\numero{28} e mostren ho vostres pels fora m\'{\i}da.

\numero{29} Car sius jaquiu / vostra barba criada

\numero{30} e laus toleu / puix ab los pels dels bra\c{c}os

\numero{31} poran sen fer / auantatjosos la\c{c}os

\numero{32} prenint Perdiu / o Tortra, o Bec\'{a}da.

\end{estrofa}



\begin{estrofa}

\numero{33} Quant oyrey / alcauota prouada

\numero{34} responeu tost / que per vos ho diran

\numero{35} e puix per nom propi vos cridaran

\numero{36} ja nous mostreu / en l'hoir empatxada.

\numero{37} Interrogant / amichs e que voleu?

\numero{38} en dret d'Amor / ay res que yo fer pusca?

\numero{39} tracte semblant / james me troba cusca

\numero{40} presta sere / a quant demanareu.

\end{estrofa}
