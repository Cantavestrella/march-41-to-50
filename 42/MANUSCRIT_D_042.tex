\begin{estrofa}

\numero{1} Vos qui sabeu delatortral costum

\numero{2} esinou feu placial vos hoyr

\numero{3} quant mort litol son par sevol jaquir

\numero{4} dobres damor no beu aygua de flum

\numero{5} Ans en los clots ensutzaprimer laygua

\numero{6} nos posa may envert arbre fullat

\numero{7} mas contra\c{c}o es vostra qualitat

\numero{8} pergrandesig no cast quenvos seraygua

\end{estrofa}



\begin{estrofa}

\numero{9} E no cuydeu dona quebeus escayga

\numero{10} que puis hagues tastat lacarngentil

\numero{11} amercader lliuras vostre cos vill

\numero{12} esondret nom en Joan me pens cayga

\numero{13} Esivoleu queus nedonconexen\c{c}a

\numero{14} sa fa\c{c} es gran ablavista molt llosca

\numero{15} sos fonaments sondelagost o mosca

\numero{16} cert no merex draps vendre de floren\c{c}a

\end{estrofa}



\begin{estrofa}

\numero{17} [38v] Econexent lavostragran fallen\c{c}a

\numero{18} volgues muntar en amar caualler

\numero{19} esabent ell tot vostre fet enver

\numero{20} envos amar setengra conciencia

\numero{21} Sabent molt clar lan sutzadavida

\numero{22} prenint publich les pagues delpeccat

\numero{23} vostrecos leig perdrap es baratat

\numero{24} vostre seruir es bo sols peradida

\end{estrofa}



\begin{estrofa}

\numero{25} Eno cuydeu fillaus agues jaquida

\numero{26} vos alletant aquellavostra llet

\numero{27} carvostre cors es de veri replet

\numero{28} e mostren ho v\textit{ost}res pels fora mida

\numero{29} Car sius jaquiu vostra barba criada

\numero{30} elas toleu puisab los pels dels bra\c{c}os

\numero{31} poran ser fer auantatjososlla\c{c}os

\numero{32} prenints perdiu otortra obequada

\end{estrofa}



\begin{estrofa}

\numero{33} Quant oyreu alcauotaprouada

\numero{34} responeu tost quepervos ho diran

\numero{35} epuis pernomproprivos cridaran

\numero{36} ja nous mostreu enloyr empachada

\numero{37} Enterrogant amichs equevoleu

\numero{38} endret damor hay res queyo ferpusca

\numero{39} tracte semblant james me troba cusca

\numero{40} presta sere aquant demanareu

\end{estrofa}



\begin{estrofa}

\numero{41} Tots los qui trob / acun\c{c}ament volreu

\numero{42} en fets damor amprau na mon boy

\numero{43} ellaus fara totlo que feu amy\footnote{\textit{Corregit sobre }ami\textit{.}}

\numero{44} nos pot saber lendre\c{c} quey trobareu

\end{estrofa}
