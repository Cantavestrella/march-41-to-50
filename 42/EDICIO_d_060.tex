\poema{(d 60) [42]}\\

\begin{rubrica}

DE AMOR CANTO. LX.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} V\textsc{o}s que sabeu / dela Tortra'l cost\'{u}m

\numero{2} e si nou feu / placial vos ho\'{\i}r

\numero{3} quant mort litol / son par.sevol jaquir

\numero{4} d'obres d'Amor / no beu aygua defl\'{u}m.

\numero{5} Ans en los clots / ensutza primer la'ygua

\numero{6} nos posa may / en vert arbre fullat.

\numero{7} mas contra'\c{c}o / es vostra cal\'{\i}t\'{a}t

\numero{8} per gran desig / no cast:que'n vos se ra\'{\i}ga.

\end{estrofa}



\begin{estrofa}

\numero{9} [74v] \textparagraph\  E no cuideu / dona que be vs escaiga

\numero{10} que puis hagues / tastat la carn gentil

\numero{11} a mercader / lliur\'{a}s vostre cos vil

\numero{12} \'{e} son dret no m / en Ioan me pens que ca\'{\i}ga.

\numero{13} E si voleu / que vs ne don coneixen\c{c}a

\numero{14} sa faz es gran / ab la vista molt llosca

\numero{15} sos fonaments / son de llagost e mosca

\numero{16} \c{c}ert no mereix / drap vendre defloren\c{c}a.

\end{estrofa}



\begin{estrofa}

\numero{17} \textparagraph\  E coneixent / la vostra gran fallen\c{c}a

\numero{18} volgues muntar / en armar caualler

\numero{19} e sabent ell / tot vostre fet en ver

\numero{20} de vos amar / se fara con\c{c}ien\c{c}a.

\numero{21} Sabent molt clar / la en sutzada vida

\numero{22} prenint publich / les pagues del peccat

\numero{23} vostre cos leig / per drap es barat\'{a}t

\numero{24} vostre seruir / es bo sols pera d\'{\i}da.

\end{estrofa}



\begin{estrofa}

\numero{25} \textparagraph\  E no cuydeu / filla vs hagues iaqu\'{\i}da

\numero{26} vos alletant / aquella ab vobra\footnote{\textit{Per }vibra.} llet

\numero{27} car vostre cos / es de veri replet

\numero{28} e mostren ho / vostres pels fora mida.

\numero{29} Car sius jaquiu / vostra barba criada

\numero{30} e laus toleu / puis ab los pels dels bra\c{c}os

\numero{31} poran sen fer / auantatjosos lla\c{c}os

\numero{32} prenint Perdiu / ho Tortra o Bec\'{a}da.

\end{estrofa}



\begin{estrofa}

\numero{33} [75r] \textparagraph\  Quant oyreu / alcauota prouada

\numero{34} responeu tost / que per vos ho diran

\numero{35} e puix per nom / propri:vos cridaran

\numero{36} ja no vs mostreu / en l'hoir enpachada.

\numero{37} Interrogant / amichs e que voleu?

\numero{38} endret d'Amor / ay res que yo fer pusca?

\numero{39} tracte semblant / james me troba cusca

\numero{40} presta sere / aquant demanareu.

\end{estrofa}

\numero{41} \textparagraph\  Tots los qui tost / acun\c{c}ament volreu

\numero{42} enfets d'Amor / anprau na Monboy

\numero{43} ella vs fara / tot lo que feu ami

\numero{44} nos pot saber / l'endre\c{c} quei trobareu.
