\poema{(E 121) [42]}\\

\begin{rubrica}

Moss\textit{en }ausias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Vos qui sabeu / dela tortral costum\footnote{\textit{Al marge:} Opinio no es ///
De Ausias march esta obra.}

\numero{2} e si nol feu / placial vos oyr

\numero{3} quant mort li tol / son par se vol jaquir

\numero{4} dobres damor / ne beu aygua de flum

\numero{5} Ans en los clots / ensutza primer laygua

\numero{6} nes posa may / en vert arbre fullat

\numero{7} mas contra\c{c}o / es vostra calitat

\numero{8} per gran desig / no cast quen vos sen rayga

\end{estrofa}



\begin{estrofa}

\numero{9} [185v] E no cuydeu / dona que beus escayga

\numero{10} que puis hagues / tastat la carn gentil

\numero{11} a mercader / lliuras vostre cors vil

\numero{12} e son dret nom / en Johan me pens cayga

\numero{13} E si voleu / queus ne do conexen\c{c}a

\numero{14} sa fa\c{c} es gran / ab la vista molt llosca

\numero{15} sos fonaments / son de llagost o mosca

\numero{16} \c{c}ert no mereix / vendre draps de floren\c{c}a

\end{estrofa}



\begin{estrofa}

\numero{17} E coneixent / la vostra gran fallen\c{c}a

\numero{18} volgues muntar / en amar caualler

\numero{19} e sabent ell / tot vostre fet en ver

\numero{20} de vos amar / se dara concien\c{c}a

\numero{21} Sabent molt clar / lansutzeada vida

\numero{22} prenint publich / les pagues del pecat

\numero{23} vostre cors lleig / per drap es baratat

\numero{24} vostre seruir / es bo sols\footnote{s \textit{afegida, quasi imperceptible}.}
pera dida

\end{estrofa}



\begin{estrofa}

\numero{25} E no cuydeu / fillaus hagues jaquida

\numero{26} vos alletant / aquellab vostra llet

\numero{27} car vostre cos / es de veri replet

\numero{28} e mostren ho / vostres pels fora mida

\numero{29} Car sius jaquiu / vostra barba criada

\numero{30} e laus toleu / puis ab los pels dels bra\c{c}os

\numero{31} poran sen fer / auantatiosos lla\c{c}os

\numero{32} prenent perdius o tortra o becada

\end{estrofa}



\begin{estrofa}

\numero{33} [186r] Quant hoyreu / alcauota prouada

\numero{34} responeu tost / que per vos ho diran

\numero{35} e puis per nom / propi vos cridaran

\numero{36} ja nous mostreu / en lohir empachada

\numero{37} Interrogant / amichs e que voleu

\numero{38} en dret damor / ay res que yo fer pusca

\numero{39} tracte semblant / James me troba cusca

\numero{40} presta sere / aquant demanareu

\end{estrofa}
