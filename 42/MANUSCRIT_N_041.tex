\begin{estrofa}

\numero{1} \textparagraph\  Vos quj sabeu dela tortral costum

\numero{2} E si nou feu placial vos hoyr

\numero{3} Quant mort litol son par se uol jaquir

\numero{4} Dobres damor ne be aygua de flum

\numero{5} Ans en los clots ensutze primer laygua

\numero{6} Nes possa may envert arbre fullat

\numero{7} Mas contra co es vostra qualitat

\numero{8} Per gran desig no cast quen vos se raygua

\end{estrofa}



\begin{estrofa}

\numero{9} [83] Eno cuydey dona que be\textit{us} escaygua

\numero{10} Que pus agues tastat lacarn gentill

\numero{11} Ha mercader liuras uostre cors vill

\numero{12} E son dret nom \interlin{en} johan me pens caygua

\numero{13} E si voleu que\textit{us}ne don conexenca

\numero{14} Sa fac es gran ela vista molt losca

\numero{15} Sos fonaments son de lagost omosca

\numero{16} Cert nomerex drabs vendre de florenca

\end{estrofa}



\begin{estrofa}

\numero{17} E conaxent lavostra gran fallenca

\numero{18} Volgues muntar en amar caualler

\numero{19} E sabent ell tot vostre fet enuer

\numero{20} En vos amar se tengra concienca

\numero{21} Sabent molt clar la sutzeada vida

\numero{22} Prenent publich les pagues del peccat

\numero{23} Vostre cors leg per drap es baratat

\numero{24} Vostre seruir es bo sol per adida

\end{estrofa}



\begin{estrofa}

\numero{25} Eno cuydey filla\textit{us} agues jaquita

\numero{26} Vos aletant aquellas vostra let

\numero{27} Car vostre cors es de verj replet

\numero{28} Emostrenho vostres pels fora mida

\numero{29} Car si\textit{us} jaquju vostra barba criada

\numero{30} E la\textit{us} toleu puys ablos pels dels bracos

\numero{31} Poran se fer auantajosos lacos

\numero{32} Prenints perdi\textit{us} tortra ho cugulada

\end{estrofa}



\begin{estrofa}

\numero{33} [84] Quant hoyreu alcauota prouada

\numero{34} Responeu tost que pervos ho diran

\numero{35} E puys pernom propi vos cridaran

\numero{36} Ja no\textit{us} mostreu enloyr empatxada

\numero{37} Enterrogant amichs eque voleu

\numero{38} Endret damor voleu res que fer pusca

\numero{39} Tracte sembant yames me troba cusca

\numero{40} Presta sere aquant demanareu

\end{estrofa}
